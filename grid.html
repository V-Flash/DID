$(document).ready(function() { var owl = $('.owl-carousel'); owl.owlCarousel(); // Handle filter button click $('.filter-btn').on('click', function() { var filterValue = $(this).data('filter'); // Toggle active state of the clicked filter button $(this).toggleClass('active'); // Get the active filter buttons var activeFilters = $('.filter-btn.active'); var activeFilterValues = activeFilters.map(function() { return $(this).data('filter'); }).get(); // Filter carousel items based on the selected options if (activeFilters.length === 0) { owl.trigger('refresh.owl.carousel'); owl.find('.item').show(); } else { owl.trigger('refresh.owl.carousel'); owl.find('.item').hide(); activeFilterValues.forEach(function(filter) { owl.find('.item[data-colors="' + filter + '"]').show(); }); } }); });
